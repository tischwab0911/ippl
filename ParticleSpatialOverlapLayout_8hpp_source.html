2// Class ParticleSpatialOverlapLayout
3// Particle layout based on spatial decomposition.
4//
5// This is a specialized version of ParticleSpatialLayout, which allows
6// particles to be on multiple processes if they are close to the respective
7// region.
8//
9// After each 'time step' in a calculation, which is defined as a period
10// in which the particle positions may change enough to affect the global
11// layout, the user must call the 'update' routine, which will move
12// particles between processors, etc. After the Nth call to update, a
13// load balancing routine will be called instead. The user may set the
14// frequency of load balancing (N), or may supply a function to
15// determine if load balancing should be done or not.

41 template <typename T, unsigned Dim, class Mesh, typename... Properties>
42 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::initializeCells() {
43 const auto rank = Comm->rank();
44 const auto hLocalRegions = this->rlayout_m.gethLocalRegions();
45 for (unsigned d = 0; d < Dim; ++d) {
46 PAssert(rcutoff_m <= hLocalRegions(rank)[d].length() / 2 &&
47 "Cutoff is too big with respect to region. "
48 "Particle could be on 3 or more ranks ins one dimension");
49 }
50
51 /* precompute information of cell structure. dividing the region into cells of at least
52 * rcutoff_m length, the length of the overlap. Use std::floor to make sure the boundary
53 * cells are big enough as well.
54 */
55 totalCells_m = 1;
56 numLocalCells_m = 1;
57 for (unsigned d = 0; d < Dim; ++d) {
58 const T length = hLocalRegions(rank)[d].length();
59 const size_type nLocalCells = std::floor(length / rcutoff_m);
60 // two ghost cells, one in each direction
61 numCells_m[d] = nLocalCells + 2 * numGhostCellsPerDim_m;
62 cellWidth_m[d] = length / nLocalCells;
63 totalCells_m *= numCells_m[d];
64 numLocalCells_m *= nLocalCells;
65 }
66 numGhostCells_m = totalCells_m - numLocalCells_m;
67
68 /* calculate cell strides to compute flat index from nd-index
69 * idx = strides[0] * idx0 + strides[1] * idx1 + ...
70 */
71 std::exclusive_scan(numCells_m.begin(), numCells_m.end(), cellStrides_m.begin(), 1,
72 std::multiplies());
73
74 cellParticleCount_m = hash_type("cellParticleCount", totalCells_m);
75 cellStartingIdx_m = hash_type("cellStartingIdx", totalCells_m + 1);
76
77 // Compute cell permutation need to place ghost cells at the end to make sure only local
78 // particles are at indices 0 to numLocalParticles
79 hash_type cellPermutationForward("cell permutation forward", totalCells_m);
80 hash_type cellPermutationBackward("cell permutation backward", totalCells_m);
81
82 /* Step 1. compute prefix sum to determine where the local/ghost cells go */
83 hash_type localPrefixSum("local prefix sum", totalCells_m);
84 hash_type ghostPrefixSum("ghost prefix sum", totalCells_m);
85 const auto& numCells = numCells_m;
86
87 Kokkos::parallel_scan(
88 "scan_local", Kokkos::RangePolicy(0, totalCells_m),
89 KOKKOS_LAMBDA(const size_type i, size_type& update, const bool final) {
90 const size_type val = isLocalCellIndex(toCellIndex(i, numCells), numCells);
91 if (final) {
92 localPrefixSum(i) = update;
93 }
94 update += val;
95 });
96 Kokkos::parallel_scan(
97 "scan_ghost", Kokkos::RangePolicy(0, totalCells_m),
98 KOKKOS_LAMBDA(const size_type i, size_type& update, const bool final) {
99 const size_type val = !isLocalCellIndex(toCellIndex(i, numCells), numCells);
100 if (final) {
101 ghostPrefixSum(i) = update;
102 }
103 update += val;
104 });
105
106 /* Step 2. assign the cells at the correct locations of the permutations */
107 const auto numLocalCells = numLocalCells_m;
108 Kokkos::parallel_for(
109 "assign_permutations", Kokkos::RangePolicy<>(0, totalCells_m),
110 KOKKOS_LAMBDA(const size_type i) {
111 if (const auto cellIdx = toCellIndex(i, numCells);
112 isLocalCellIndex(cellIdx, numCells)) {
113 const size_type localIdx = localPrefixSum(i);
114 cellPermutationForward(i) = localIdx;
115 cellPermutationBackward(localIdx) = i;
116 } else {
117 const size_type ghostIdx = numLocalCells + ghostPrefixSum(i);
118 cellPermutationForward(i) = ghostIdx;
119 cellPermutationBackward(ghostIdx) = i;
120 }
121 });
122
123 cellPermutationForward_m = cellPermutationForward;
124 cellPermutationBackward_m = cellPermutationBackward;
125 }

140 template <typename T, unsigned Dim, class Mesh, typename... Properties>
141 template <class ParticleContainer>
142 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::createPeriodicGhostParticles(
143 ParticleContainer& pc) {
144 static IpplTimings::TimerRef timer = IpplTimings::getTimer("createPeriodicGhostParticles");
145 IpplTimings::startTimer(timer);
146 /* periodic boundary conditions come in pairs. Thus collect whether each dimension is
147 * subject to periodic boundary conditions
148 */
149 Vector_t<bool, Dim> periodic;
150 for (unsigned d = 0; d < Dim; ++d) {
151 periodic[d] = this->getParticleBC()[2 * d];
152 }
153 // no need to create periodic ghost particles if no dimension is periodic
154 if (!std::any_of(periodic.begin(), periodic.end(), [](auto bc) {
155 return bc == BC::PERIODIC;
156 })) {
157 return;
158 }
159
160 const auto& globalRegion = this->rlayout_m.getDomain();
161 const auto overlap = rcutoff_m;
162 const auto numLoc = pc.getLocalNum();
163 const auto positions = pc.R.getView();
164
165 /* Step 1. Determine all particles which are close to the global domain boundary */
166 size_type numBoundaryParticles = 0;
167 Kokkos::parallel_reduce(
168 "count boundary particles", numLoc,
169 KOKKOS_LAMBDA(const size_t& i, size_type& sum) {
170 if (isCloseToBoundary(positions(i), globalRegion, periodic, overlap)) {
171 ++sum;
172 }
173 },
174 Kokkos::Sum<size_type>(numBoundaryParticles));
175 if (numBoundaryParticles == 0) {
176 return;
177 }
178
179 /* Step 2. compute prefix sum to collect boundary particle indices */
180 hash_type boundaryIndices("boundaryIndices", numBoundaryParticles);
181 Kokkos::parallel_scan(
182 "count boundary particles", numLoc,
183 KOKKOS_LAMBDA(const size_t& i, size_type& sum, bool final) {
184 if (isCloseToBoundary(positions(i), globalRegion, periodic, overlap)) {
185 if (final) {
186 boundaryIndices(sum) = i;
187 }
188 ++sum;
189 }
190 });
191
192 /* Step 3. copy given particles and all its attributes */
193 detail::runForAllSpaces([&]<typename MemorySpace>() {
194 size_t numAttributesInSpace = 0;
195 pc.template forAllAttributes<MemorySpace>([&]<typename Attribute>(Attribute&) {
196 ++numAttributesInSpace;
197 });
198 if (numAttributesInSpace == 0) {
199 return;
200 }
201
202 auto boundaryIndicesMirror =
203 Kokkos::create_mirror_view_and_copy(MemorySpace(), boundaryIndices);
204 pc.template forAllAttributes<MemorySpace>([&]<typename Attribute>(Attribute& att) {
205 att->internalCopy(boundaryIndicesMirror);
206 });
207 });
208
209 /* Step 4. set the position of the copied particles to their periodic image */
210 for (unsigned d = 0; d < Dim; ++d) {
211 if (!periodic[d]) {
212 continue;
213 }
214 const auto min = globalRegion[d].min();
215 const auto length = globalRegion[d].length();
216 const auto middle = min + length / 2;
217 Kokkos::parallel_for(
218 "correct positions",
219 // the copied particles are appended at the end of the attributes
220 Kokkos::RangePolicy<position_execution_space>(numLoc,
221 numLoc + numBoundaryParticles),
222 KOKKOS_LAMBDA(const size_t& i) {
223 positions(i)[d] += positions(i)[d] > middle ? -length : length;
224 });
225 }
226
227 /* make sure following functions (particleExchange and buildCells) know about the ghost
228 * particles. They will not stay pc's particle as their positions are outside the global
229 * domain but will be ghost particles.
230 */
231 pc.setLocalNum(numLoc + numBoundaryParticles);
232 IpplTimings::stopTimer(timer);
233 }

235 template <typename T, unsigned Dim, class Mesh, typename... Properties>
236 template <class ParticleContainer>
237 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::particleExchangeNew(
238 ParticleContainer& pc) {
239 /* Apply Boundary Conditions */
240 static IpplTimings::TimerRef ParticleBCTimer = IpplTimings::getTimer("particleBC");
241 IpplTimings::startTimer(ParticleBCTimer);
242 this->applyBC(pc.R, this->rlayout_m.getDomain());
243 createPeriodicGhostParticles(pc);
244 IpplTimings::stopTimer(ParticleBCTimer);
245
246 /* Update Timer for the rest of the function */
247 static IpplTimings::TimerRef ParticleUpdateTimer = IpplTimings::getTimer("updateParticle");
248 IpplTimings::startTimer(ParticleUpdateTimer);
249
250 int nRanks = Comm->size();
251 if (nRanks < 2) {
252 return;
253 }
254
255 /* particle MPI exchange:
256 * 1. figure out which particles need to go where -> locateParticles(...)
257 * 2. fill send buffer and send particles
258 * 3. delete invalidated particles
259 * 4. receive particles
260 */
261
262 // 1. figure out which particles need to go where -> locateParticles(...) ============= //
263
264 static IpplTimings::TimerRef locateTimer = IpplTimings::getTimer("locateParticles");
265 IpplTimings::startTimer(locateTimer);
266
267 /* Rank-local number of particles */
268 size_type localnum = pc.getLocalNum();
269
270 /* particleRanks are the indices correspond to the indices of the local particles,
271 * the values correspond to the ranks to which the particles need to be sent note that the
272 * size is not known yet as particles may belong to multiple ranks
273 * particleRankOffsets are the offsets of each particle as a particle
274 * can be sent to multiple ranks
275 */
276 locate_type particleRanks("particles' MPI ranks");
277 locate_type particleRankOffsets("particles' MPI rank offsets", localnum + 1);
278
279 /* The indices are the indices of the particles,
280 * the boolean values describe whether the particle has left the current rank
281 * 0 --> particle valid (inside current rank)
282 * 1 --> particle invalid (left rank)
283 */
284 bool_type invalidParticles("validity of particles", localnum);
285
286 /* The indices are the MPI ranks,
287 * the values are the number of particles are sent to that rank from myrank
288 */
289 locate_type rankSendCount_dview("rankSendCount Device", nRanks);
290
291 /* The indices have no particular meaning,
292 * the values are the MPI ranks to which we need to send
293 */
294 locate_type destinationRanks_dview("destinationRanks Device", nRanks);
295
296 /* nInvalid is the number of invalid particles
297 * nDestinationRanks is the number of MPI ranks we need to send to
298 */
299 const auto [nInvalid, nDestinationRanks] =
300 locateParticles(pc, particleRanks, particleRankOffsets, invalidParticles,
301 rankSendCount_dview, destinationRanks_dview);
302
303 /* Host space copy of rankSendCount_dview */
304 auto rankSendCount_hview =
305 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace(), rankSendCount_dview);
306
307 /* Host Space copy of destinationRanks_dview */
308 auto destinationRanks_hview =
309 Kokkos::create_mirror_view_and_copy(Kokkos::HostSpace(), destinationRanks_dview);
310 Kokkos::fence();
311
312 IpplTimings::stopTimer(locateTimer);
313
314 // 2. fill send buffer and send particles =============================================== //
315
316 // 2.1 Remote Memory Access window for one-sided communication
317
318 static IpplTimings::TimerRef preprocTimer = IpplTimings::getTimer("sendPreprocess");
319 IpplTimings::startTimer(preprocTimer);
320
321 std::fill(this->nRecvs_m.begin(), this->nRecvs_m.end(), 0);
322
323 this->window_m.fence(0);
324
325 // Prepare RMA window for the ranks we need to send to
326 for (size_t ridx = 0; ridx < nDestinationRanks; ridx++) {
327 int rank = destinationRanks_hview[ridx];
328 if (rank == Comm->rank()) {
329 // we do not need to send to ourselves
330 continue;
331 }
332 this->window_m.template put<size_type>(rankSendCount_hview(rank), rank, Comm->rank());
333 }
334 this->window_m.fence(0);
335
336 IpplTimings::stopTimer(preprocTimer);
337
338 // 2.2 Particle Sends
339
340 static IpplTimings::TimerRef sendTimer = IpplTimings::getTimer("particleSend");
341 IpplTimings::startTimer(sendTimer);
342
343 std::vector<MPI_Request> requests(0);
344
345 int tag = Comm->next_tag(mpi::tag::P_SPATIAL_LAYOUT, mpi::tag::P_LAYOUT_CYCLE);
346
347 for (size_t ridx = 0; ridx < nDestinationRanks; ridx++) {
348 int rank = destinationRanks_hview[ridx];
349 if (rank == Comm->rank()) {
350 continue;
351 }
352 hash_type hash("hash", rankSendCount_hview(rank));
353 fillHash(rank, particleRanks, particleRankOffsets, hash);
354 pc.sendToRank(rank, tag, requests, hash);
355 }
356
357 IpplTimings::stopTimer(sendTimer);
358
359 // 3. Internal destruction of invalid particles ======================================= //
360
361 static IpplTimings::TimerRef destroyTimer = IpplTimings::getTimer("particleDestroy");
362 IpplTimings::startTimer(destroyTimer);
363
364 pc.internalDestroy(invalidParticles, nInvalid);
365 Kokkos::fence();
366
367 IpplTimings::stopTimer(destroyTimer);
368
369 // 4. Receive Particles ================================================================ //
370
371 static IpplTimings::TimerRef recvTimer = IpplTimings::getTimer("particleRecv");
372 IpplTimings::startTimer(recvTimer);
373
374 for (int rank = 0; rank < nRanks; ++rank) {
375 if (this->nRecvs_m[rank] > 0) {
376 pc.recvFromRank(rank, tag, this->nRecvs_m[rank]);
377 }
378 }
379 IpplTimings::stopTimer(recvTimer);
380
381 IpplTimings::startTimer(sendTimer);
382
383 if (requests.size() > 0) {
384 MPI_Waitall(requests.size(), requests.data(), MPI_STATUSES_IGNORE);
385 }
386 IpplTimings::stopTimer(sendTimer);
387
388 IpplTimings::stopTimer(ParticleUpdateTimer);
389 }

534 template <typename T, unsigned Dim, class Mesh, typename... Properties>
535 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::fillHash(
536 int rank, const locate_type& ranks, const locate_type& offsets, hash_type& hash) {
537 /* Compute the prefix sum and fill the hash
538 */
539 using policy_type = Kokkos::RangePolicy<position_execution_space>;
540 Kokkos::parallel_scan(
541 "ParticleSpatialLayout::fillHash()", policy_type(0, offsets.extent(0) - 1),
542 KOKKOS_LAMBDA(const size_t i, int& idx, const bool final) {
543 /* Check if this particle belongs to our target rank. Ranks of particle i are stored
544 * from offsets(i) to offsets(i + 1)
545 */
546 bool belongs_to_rank = false;
547 const size_t start_rank_idx = offsets(i);
548 const size_t end_rank_idx = offsets(i + 1);
549
550 for (size_t rank_idx = start_rank_idx; rank_idx < end_rank_idx; ++rank_idx) {
551 if (ranks(rank_idx) == rank) {
552 belongs_to_rank = true;
553 break; // Found it, no need to continue
554 }
555 }
556
557 if (final && belongs_to_rank) {
558 hash(idx) = i;
559 }
560
561 if (belongs_to_rank) {
562 idx += 1;
563 }
564 });
565 Kokkos::fence();
566 }

570 template <typename T, unsigned Dim, class Mesh, typename... Properties>
571 typename ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::locate_type
572 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getFlatNeighbors(
573 const neighbor_list& neighbors) const {
574 /* collect all neighbors in a set. set instead of vector as with small number of ranks and
575 * periodic boundary conditions the same rank can be in the neighbor list multiple times.
576 */
577 std::set<int> neighborSet;
578 for (const auto& componentNeighbors : neighbors) {
579 for (const auto& nrank : componentNeighbors) {
580 neighborSet.insert(nrank);
581 }
582 }

585 locate_type flatNeighbors("Nearest neighbors IDs", neighborSet.size());
586 auto hostMirror = Kokkos::create_mirror_view(flatNeighbors);
587
588 size_type i = 0;
589 for (const auto& neighbor : neighborSet) {
590 hostMirror(i) = neighbor;
591 ++i;
592 }
593
594 Kokkos::deep_copy(flatNeighbors, hostMirror);
595 Kokkos::fence();
596 return flatNeighbors;

599 /* Helper function to get non-neighboring ranks
600 */
601 template <typename T, unsigned Dim, class Mesh, typename... Properties>
602 typename ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::locate_type
603 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getNonNeighborRanks(
604 const locate_type& neighbors_view) const {
605 // Create a view of all non-neighbor ranks. make sure to exclude own its rank
606 const auto numNonNeighborRanks = Comm->size() - neighbors_view.extent(0) - 1;
607 locate_type nonNeighborRanks("Non Neighbor Ranks", numNonNeighborRanks);
608 if (numNonNeighborRanks == 0) {
609 return nonNeighborRanks;
610 }
611
612 // Step 1. Mark all non-neighbor ranks, by removing all neighbors and self
613 const auto total_ranks = Comm->rank();
614 bool_type is_remaining("is_remaining", total_ranks);
615 Kokkos::deep_copy(is_remaining, true);
616 Kokkos::fence();
617
618 const auto myRank = Comm->rank();
619 Kokkos::parallel_for(
620 "mark_comm_ranks", Kokkos::RangePolicy(myRank, myRank + 1),
621 KOKKOS_LAMBDA(const size_t& i) { is_remaining(i) = false; });
622 Kokkos::parallel_for(
623 "mark_comm_ranks", neighbors_view.extent(0),
624 KOKKOS_LAMBDA(const size_t& i) { is_remaining(neighbors_view(i)) = false; });
625 Kokkos::fence();
626
627 // Step 2. Fill remaining ranks
628 Kokkos::View<size_type> counter("counter");
629 Kokkos::parallel_for(
630 "fill_remaining", total_ranks, KOKKOS_LAMBDA(const size_t& i) {
631 if (is_remaining(i)) {
632 const size_type idx = Kokkos::atomic_fetch_inc(&counter());
633 nonNeighborRanks(idx) = i;
634 }
635 });
636 return nonNeighborRanks;

639 template <typename T, unsigned Dim, class Mesh, typename... Properties>
640 template <typename ParticleContainer>
641 std::pair<detail::size_type, detail::size_type>
642 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::locateParticles(
643 const ParticleContainer& pc, locate_type& ranks, locate_type& rankOffsets,
644 bool_type& invalid, locate_type& nSends_dview, locate_type& sends_dview) const {
645 const auto positions = pc.R.getView();
646 const auto regions = this->rlayout_m.getdLocalRegions();
647 const auto myRank = Comm->rank();
648 const auto localNum = pc.getLocalNum();
649 const T overlap = rcutoff_m;
650
653 locate_type counts("counts", localNum);
654 locate_type outsideIds("Particles outside of neighborhood", localNum);
655
657 size_type outsideCount = 0;
659 size_type invalidCount = 0;
660
662 locate_type neighbors_view = getFlatNeighbors(this->flayout_m.getNeighbors());
663
664 /* red_val: Used to reduce both the number of invalid particles and the number of particles
665 * outside the neighborhood (Kokkos::parallel_scan doesn't allow multiple reduction
666 * values, so we use the helper class increment_type). First element updates InvalidCount,
667 * second one updates outsideCount.
668 */
669 increment_type red_val;
670 red_val.init();
671
678 static IpplTimings::TimerRef neighborSearch = IpplTimings::getTimer("neighborSearch");
679 IpplTimings::startTimer(neighborSearch);
680
681 /* First Pass: count the numbers of neighbor ranks (including self) a particle belongs to */
682 Kokkos::parallel_for(
683 "ParticleSpatialLayout::locateParticles()", localNum, KOKKOS_LAMBDA(const size_t& i) {
684 const bool inCurr = positionInRegion(positions(i), regions(myRank), overlap);
685
686 size_type count = inCurr;
687 invalid(i) = !inCurr;
688 // Count neighboring regions
689 for (size_type j = 0; j < neighbors_view.extent(0); ++j) {
690 const size_type rank = neighbors_view(j);
691 if (positionInRegion(positions(i), regions(rank), overlap)) {
692 ++count;
693 }
694 }
695
696 counts(i) = count;
697 });
698 Kokkos::fence();
699
700 /* Second Pass: collect number of particles outside this ranks region and the indices of the
701 * respective particles. Note that in comparison to ParticleSpatialLayout::locateParticles()
702 * particle can be in multiple ranks. For this reason if a particle is in a neighbor but not
703 * here, then we still need to search in all ranks as there is no way to get second
704 * neighbors.
705 */
706 Kokkos::parallel_scan(
707 "count_outside", localNum,
708 KOKKOS_LAMBDA(const size_type i, increment_type& val, const bool final) {
709 const bool inCurr = !invalid(i);
710 if (final && !inCurr) {
711 outsideIds(val.count[1]) = i;
712 }
713 bool increment[2] = {invalid(i), !inCurr};
714 val += increment;
715 },
716 red_val);
717 Kokkos::fence();
718
719 invalidCount = red_val.count[0];
720 outsideCount = red_val.count[1];
721
722 IpplTimings::stopTimer(neighborSearch);
723
725 static IpplTimings::TimerRef nonNeighboringParticles =
726 IpplTimings::getTimer("nonNeighboringParticles");
727 IpplTimings::startTimer(nonNeighboringParticles);
728
729 using mdrange_type = Kokkos::MDRangePolicy<Kokkos::Rank<2>, position_execution_space>;
730
731 /* Count the number of non-neighbor ranks each particle belongs to */
732 locate_type outsideCounts("counts of outside neighbors", outsideCount);
733 locate_type nonNeighborsView = getNonNeighborRanks(neighbors_view);
734 if (outsideCount > 0 && nonNeighborsView.extent(0) > 0) {
735 Kokkos::deep_copy(outsideCounts, 0);
736 Kokkos::parallel_for(
737 "ParticleSpatialLayout::leftParticles()",
738 mdrange_type({0, 0}, {outsideCount, nonNeighborsView.extent(0)}),
739 KOKKOS_LAMBDA(const size_t i, const size_type j) {
741 const size_type pId = outsideIds(i);
742 const auto rank = nonNeighborsView(j);
743
745 if (positionInRegion(positions(pId), regions(rank), overlap)) {
746 Kokkos::atomic_increment(&outsideCounts(i));
747 }
748 });
749 Kokkos::fence();
750 Kokkos::parallel_for(
751 "ParticleSpatialLayout::leftParticles()", outsideCount,
752 KOKKOS_LAMBDA(const size_t& i) { counts(outsideIds(i)) += outsideCounts(i); });
753 Kokkos::fence();
754 }
755 IpplTimings::stopTimer(nonNeighboringParticles);
756
757 IpplTimings::startTimer(neighborSearch);
758 /* prefix sum for particle rank offsets */
759 Kokkos::deep_copy(Kokkos::subview(rankOffsets, 0), 0);
760 Kokkos::parallel_scan(
761 "ParticleSpatialLayout::locateParticles()", localNum,
762 KOKKOS_LAMBDA(const size_t i, size_type& localSum, const bool final) {
763 const auto count_i = counts(i);
764 if (final) {
765 rankOffsets(i + 1) = localSum + count_i;
766 }
767 localSum += count_i;
768 });
769 Kokkos::fence();
770
771 /* Get total number of assignments for allocation from the last entry of offsets */
772 auto total_assignments = Kokkos::create_mirror_view(Kokkos::subview(rankOffsets, localNum));
773 Kokkos::deep_copy(total_assignments, Kokkos::subview(rankOffsets, localNum));
774 Kokkos::fence();
775 Kokkos::resize(ranks, total_assignments());
776
777 /* Last Pass: fill the rank data */
778 Kokkos::parallel_for(
779 "ParticleSpatialLayout::locateParticles()", localNum, KOKKOS_LAMBDA(const size_t& i) {
780 const size_t offset = rankOffsets(i);
781 size_type local_count = 0;
782 if (positionInRegion(positions(i), regions(myRank), overlap)) {
783 ranks(offset) = myRank;
784 local_count++;
785 }
786 for (size_t j = 0; j < neighbors_view.extent(0); ++j) {
787 const auto nRank = neighbors_view(j);
788 if (positionInRegion(positions(i), regions(nRank), overlap)) {
789 ranks(offset + local_count) = nRank;
790 local_count++;
791 }
792 }
793 });
794 Kokkos::fence();
795
796 IpplTimings::stopTimer(neighborSearch);
797
798 /* Last Pass: add the data of the outside particles to the ranks */
799 IpplTimings::startTimer(nonNeighboringParticles);
800 if (outsideCount > 0) {
801 Kokkos::parallel_for(
802 "ParticleSpatialLayout::leftParticles()", outsideCount,
803 KOKKOS_LAMBDA(const size_t& i) {
805 const size_type pId = outsideIds(i);
806 const size_type offset = rankOffsets(pId) + counts(pId);
807 for (size_t local_count = 0, j = 0; j < nonNeighborsView.extent(0); ++j) {
808 const auto rank = nonNeighborsView(j);
809 if (positionInRegion(positions(i), regions(rank), overlap)) {
810 ranks(offset + local_count) = rank;
811 local_count++;
812 }
813 }
814 });
815 Kokkos::fence();
816 }
817 IpplTimings::stopTimer(nonNeighboringParticles);
818
819 /* compute the number of sends to all ranks */
820 Kokkos::deep_copy(nSends_dview, 0);
821 Kokkos::parallel_for(
822 "Calculate nSends", Kokkos::RangePolicy<size_t>(0, ranks.extent(0)),
823 KOKKOS_LAMBDA(const size_t i) {
824 size_type rank = ranks(i);
825 Kokkos::atomic_increment(&nSends_dview(rank));
826 });
827 Kokkos::fence();
828
829 /* compute the ranks to send to and the number of ranks to send to*/
830 Kokkos::View<size_type> rankSends("Number of Ranks we need to send to");
831 Kokkos::parallel_for(
832 "Calculate sends", Kokkos::RangePolicy<size_t>(0, nSends_dview.extent(0)),
833 KOKKOS_LAMBDA(const size_t rank) {
834 if (nSends_dview(rank) != 0) {
835 size_type index = Kokkos::atomic_fetch_inc(&rankSends());
836 sends_dview(index) = rank;
837 }
838 });
839 Kokkos::fence();
840 size_type temp;
841 Kokkos::deep_copy(temp, rankSends);
842 Kokkos::fence();
843
844 return {invalidCount, temp};
845 }

847 template <typename T, unsigned Dim, class Mesh, typename... Properties>
848 template <typename ParticleContainer>
849 detail::size_type ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::locateParticles(
850 const ParticleContainer& pc, locate_type& ranks, locate_type& rankOffsets,
851 bool_type& invalid) const {
852 const auto positions = pc.R.getView();
853 const auto regions = this->rlayout_m.getdLocalRegions();
854 const size_type numLoc = pc.getLocalNum();
855 const auto overlap = rcutoff_m;
856
857 // First pass: count assignments per particle
858 locate_type counts("counts", numLoc);
859
860 auto range_policy = Kokkos::RangePolicy<position_execution_space>(0, numLoc);
861 Kokkos::parallel_for(
862 "count_assignments", range_policy, KOKKOS_LAMBDA(size_t i) {
863 int count = 0;
864 for (size_t j = 0; j < regions.extent(0); ++j) {
865 if (positionInRegion(positions(i), regions(j), overlap)) {
866 count++;
867 }
868 }
869 counts(i) = count;
870 });
871
872 Kokkos::fence();
873
874 // Compute prefix sum for offsets
875 // Step 2: Compute offsets using parallel_scan
876
877 locate_type offsets("offets", numLoc + 1);
878 Kokkos::deep_copy(Kokkos::subview(offsets, 0), 0);
879
880 // Compute exclusive prefix sum (each offset points to start of particle's data)
881 Kokkos::parallel_scan(
882 "compute_offsets", range_policy,
883 KOKKOS_LAMBDA(const size_t i, int& localSum, const bool final) {
884 const int count_i = counts(i);
885 if (final) {
886 offsets(i + 1) = localSum + count_i;
887 }
888 localSum += count_i;
889 });
890 Kokkos::fence();
891
892 rankOffsets = offsets;
893
894 // Get total number of assignments for allocation
895 auto total_assignments = Kokkos::create_mirror_view(Kokkos::subview(offsets, numLoc));
896 Kokkos::deep_copy(total_assignments, Kokkos::subview(offsets, numLoc));
897
898 locate_type rank_data("rank_data", total_assignments());
899
900 const size_type myRank = Comm->rank();
901
902 // Second pass: fill the data
903 size_type invalidCount = 0;
904 Kokkos::parallel_reduce(
905 "fill_assignments", range_policy,
906 KOKKOS_LAMBDA(size_t i, size_t& count) {
907 size_t offset = rankOffsets(i);
908 int local_count = 0;
909 for (size_t j = 0; j < regions.extent(0); ++j) {
910 bool xyz_bool = positionInRegion(positions(i), regions(j), overlap);
911 if (xyz_bool) {
912 rank_data(offset + local_count) = j;
913 local_count++;
914 }
915
916 if (j == myRank) {
917 invalid(i) = !xyz_bool;
918 count += !xyz_bool;
919 }
920 }
921 },
922 Kokkos::Sum<size_type>(invalidCount));
923 Kokkos::fence();
924
925 ranks = rank_data;
926
927 assert(invalidCount <= numLoc);
928
929 return invalidCount;
930 }

987 template <class ParticleContainer>
988 void ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::buildCells(
989 ParticleContainer& pc) {
990 static IpplTimings::TimerRef cellBuildTimer = IpplTimings::getTimer("cellBuildTimer");
991 IpplTimings::startTimer(cellBuildTimer);
992
993 // get local variables of all necessary data as needed for the Kokkos parallel loops
994 const auto rank = Comm->rank();
995 const size_type numLoc = pc.getLocalNum();
996 const auto positions = pc.R.getView();
997 const auto totalCells = totalCells_m;
998 const auto numLocalCells = numLocalCells_m;
999 const auto localRegion = this->rlayout_m.gethLocalRegions()(rank);
1000 const auto& cellWidth = cellWidth_m;
1001 const auto cellStrides = cellStrides_m;
1002 const auto cellPermutationForward = cellPermutationForward_m;
1003
1004 using int_type = typename particle_neighbor_list_type::value_type;
1005
1006 // allocate required (temporary) Kokkos views
1007 hash_type cellIndex("cellIndex", numLoc);
1008 hash_type cellParticleCount = cellParticleCount_m;
1009 hash_type cellStartingIdx = cellStartingIdx_m;
1010 hash_type cellCurrentIdx("cellCurrentIdx", totalCells + 1);
1011
1012 using range_policy = Kokkos::RangePolicy<position_execution_space>;
1013
1014 /* Step 1. calculate cell index for each particle and keep track of how many particles are
1015 * in each cell
1016 */
1017 Kokkos::deep_copy(cellParticleCount, 0);
1018 Kokkos::parallel_for(
1019 "CalcCellIndices", range_policy(0, numLoc), KOKKOS_LAMBDA(const size_t& i) {
1020 const auto locCellIndex = getCellIndex(positions(i), localRegion, cellWidth);
1021 const auto locCellIndexFlat =
1022 toFlatCellIndex(locCellIndex, cellStrides, cellPermutationForward);
1023 Kokkos::atomic_increment(&cellParticleCount(locCellIndexFlat));
1024 cellIndex(i) = locCellIndexFlat;
1025 });
1026 Kokkos::fence();
1027
1028 /* Step 2. compute starting indices for each cell from the counts */
1029 Kokkos::parallel_scan(
1030 "CalcStartingIndices", range_policy(0, totalCells),
1031 KOKKOS_LAMBDA(const size_t i, int_type& localSum, bool isFinal) {
1032 if (isFinal) {
1033 cellStartingIdx(i) = localSum;
1034 }
1035 localSum += cellParticleCount(i);
1036 });
1037 /* set last position of cell staring index to numLoc*/
1038 Kokkos::deep_copy(
1039 Kokkos::subview(cellStartingIdx, Kokkos::make_pair(totalCells, totalCells + 1)),
1040 numLoc);
1041 Kokkos::fence();
1042
1043 Kokkos::deep_copy(cellCurrentIdx, cellStartingIdx);
1044 Kokkos::fence();
1045
1046 hash_type newIndex("newIndex", numLoc);
1047 hash_type newCellIndex("cellIndex", numLoc);
1048
1049 /* Step 3. compute new indices for the particles such that they are sorted according to
1050 * cellStaringIdx and sort cell indices of the particles in tandem
1051 */
1052 Kokkos::parallel_for(
1053 "Calculate new Indices", range_policy(0, numLoc), KOKKOS_LAMBDA(const size_type& i) {
1054 const auto locCellIndex = cellIndex(i);
1055 const size_type newIdx = Kokkos::atomic_fetch_inc(&cellCurrentIdx(locCellIndex));
1056 newIndex(i) = newIdx;
1057 newCellIndex(newIdx) = locCellIndex;
1058 });
1059 Kokkos::fence();
1060
1061 /* Step 4. Sort all particles (and all their attributes) according to then new indices
1062 * (maybe there is a better solution)
1063 */
1064 detail::runForAllSpaces([&]<typename MemorySpace>() {
1065 size_t num_attributes_in_space = 0;
1066 pc.template forAllAttributes<MemorySpace>([&]<typename Attribute>(Attribute&) {
1067 ++num_attributes_in_space;
1068 });
1069 if (num_attributes_in_space == 0) {
1070 return;
1071 }
1072
1073 auto newIndexMirror = Kokkos::create_mirror_view_and_copy(MemorySpace(), newIndex);
1074 pc.template forAllAttributes<MemorySpace>([&]<typename Attribute>(Attribute& att) {
1075 att->applyPermutation(newIndexMirror);
1076 });
1077 });
1078 Kokkos::fence();
1079
1080 /* Step 5. set local number of particles (excluding ghost particles) is the value of
1081 * cellStartingIdx at index numLocalCells*/
1082 auto numLocalParticles =
1083 Kokkos::create_mirror_view(Kokkos::subview(cellStartingIdx, numLocalCells));
1084 Kokkos::deep_copy(numLocalParticles, Kokkos::subview(cellStartingIdx, numLocalCells));
1085
1086 numLocalParticles_m = numLocalParticles();
1087 pc.setLocalNum(numLocalParticles_m);
1088
1089 /* store the new cell indices */
1090 cellIndex_m = newCellIndex;
1091
1092 IpplTimings::stopTimer(cellBuildTimer);
1093 }

1095 template <typename T, unsigned Dim, class Mesh, typename... Properties>
1096 KOKKOS_INLINE_FUNCTION constexpr
1097 typename ParticleSpatialOverlapLayout<T, Dim, Mesh,
1098 Properties...>::cell_particle_neighbor_list_type
1099 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getCellNeighbors(
1100 const CellIndex_t& cellIndex, const Vector_t<size_type, Dim>& cellStrides,
1101 const hash_type& cellPermutationForward) {
1102 /* Generate all 3^Dim combinations of offsets (-1, 0, +1) for each dimension by using
1103 * "base-3" representation of neighbor index. in base-3 each digit is 0, 1, 2 subtracting
1104 * one leads to the desired offsets -1, 0, +1.
1105 */
1106 constexpr size_type numNeighbors = detail::countHypercubes(Dim);
1107 cell_particle_neighbor_list_type neighborIndices;
1108 for (size_type neighborIdx = 0; neighborIdx < numNeighbors; ++neighborIdx) {
1109 index_t temp = neighborIdx;
1110
1111 /* extract the offsets from the base-3 representation of the neighborIdx */
1112 auto neighborCellIndex = cellIndex;
1113 for (unsigned d = 0; d < Dim; ++d) {
1114 neighborCellIndex(d) += (temp % 3) - 1;
1115 temp /= 3;
1116 }
1117
1118 neighborIndices[neighborIdx] =
1119 toFlatCellIndex(neighborCellIndex, cellStrides, cellPermutationForward);
1120 }
1121 return neighborIndices;
1122 }

1133 template <typename T, unsigned Dim, class Mesh, typename... Properties>
1134 KOKKOS_FUNCTION
1135 typename ParticleSpatialOverlapLayout<T, Dim, Mesh,
1136 Properties...>::particle_neighbor_list_type
1137 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getParticleNeighbors(
1138 const vector_type& pos, const ParticleNeighborData& particleNeighborData) {
1139 /* get the cell index corresponding to pos */
1140 const auto locCellIndex =
1141 getCellIndex(pos, particleNeighborData.region, particleNeighborData.cellStrides,
1142 particleNeighborData.cellWidth);
1143 constexpr size_type numNeighbors = detail::countHypercubes(Dim);
1144
1145 /* cell neighbors */
1146 const auto neighbors = getCellNeighbors(locCellIndex, particleNeighborData.cellStrides,
1147 particleNeighborData.cellPermutationForward);
1148
1149 /* Get sizes of the cell neighbors and total particle neighbors */
1150 size_type totalParticleInNeighbors = 0;
1151 size_type maxParticleInNeighbors = 0;
1152
1153 Kokkos::Array<typename hash_type::value_type, numNeighbors> neighborSizes;
1154 Kokkos::Array<typename hash_type::value_type, numNeighbors> neighborOffsets;
1155 // #pragma unroll
1156 for (size_type neighborIdx = 0; neighborIdx < numNeighbors; ++neighborIdx) {
1157 auto n = particleNeighborData.cellParticleCount(neighbors[neighborIdx]);
1158 neighborSizes[neighborIdx] = n;
1159 maxParticleInNeighbors = std::max<size_type>(n, maxParticleInNeighbors);
1160 neighborOffsets[neighborIdx] = totalParticleInNeighbors;
1161 totalParticleInNeighbors += n;
1162 }
1163
1164 /* Collect the neighbor particles from all cell neighbors */
1165 particle_neighbor_list_type neighborList("Neighbor list", totalParticleInNeighbors);
1166
1167 using mdrange_policy = Kokkos::MDRangePolicy<Kokkos::Rank<2>, position_execution_space>;
1168 Kokkos::parallel_for(
1169 "collect neighbors", mdrange_policy({0, 0}, {numNeighbors, maxParticleInNeighbors}),
1170 KOKKOS_LAMBDA(const size_type& i, const size_type& j) {
1171 if (j < neighborSizes[i]) {
1172 neighborList(neighborOffsets[i] + j) =
1173 particleNeighborData.cellStartingIdx(neighbors[i]) + j;
1174 }
1175 });
1176 Kokkos::fence();
1177
1178 return neighborList;
1179 }

1181 template <typename T, unsigned Dim, class Mesh, typename... Properties>
1182 KOKKOS_FUNCTION
1183 typename ParticleSpatialOverlapLayout<T, Dim, Mesh,
1184 Properties...>::particle_neighbor_list_type
1185 ParticleSpatialOverlapLayout<T, Dim, Mesh, Properties...>::getParticleNeighbors(
1186 index_t particleIndex, const ParticleNeighborData& particleNeighborData) {
1187 // Get the cell of the particle
1188 constexpr size_type numNeighbors = detail::countHypercubes(Dim);
1189
1190 /* get the cell index corresponding to particleIndex */
1191 const auto locCellIndexFlat = particleNeighborData.cellIndex(particleIndex);
1192 const auto locCellIndex =
1193 toCellIndex(particleNeighborData.cellPermutationBackward(locCellIndexFlat),
1194 particleNeighborData.numCells);
1195
1196 /* cell neighbors */
1197 const auto neighbors = getCellNeighbors(locCellIndex, particleNeighborData.cellStrides,
1198 particleNeighborData.cellPermutationForward);
1199
1200 /* Get sizes of the cell neighbors and total particle neighbors */
1201 size_type totalParticleInNeighbors = 0;
1202 size_type maxParticleInNeighbors = 0;
1203
1204 Kokkos::Array<size_type, numNeighbors> neighborSizes;
1205 Kokkos::Array<size_type, numNeighbors + 1> neighborOffsets;
1206 // #pragma unroll
1207 for (size_type neighborIdx = 0; neighborIdx < numNeighbors; ++neighborIdx) {

1281 */

1288 neighborCellParticleOffset + j;

1316 int nRanks = Comm->size();

1318 if (nRanks < 2) {
1319 return;
1320 }
1321
1322 /* particle MPI exchange:
1323 * 1. figure out which particles need to go where
1324 * 2. fill send buffer and send particles
1325 * 3. delete invalidated particles
1326 * 4. receive particles
1327 */
1328
1329 static IpplTimings::TimerRef locateTimer = IpplTimings::getTimer("locateParticles");
1330 IpplTimings::startTimer(locateTimer);
1331 size_type localnum = pc.getLocalNum();
1332
1333 // 1st step
1334
1335 /* the values specify the rank where
1336 * the particle with that index should go
1337 */
1338 locate_type_nd ranks("MPI ranks", localnum);
1339
1340 /* 0 --> particle valid
1341 * 1 --> particle invalid
1342 */

1344
1345 size_type invalidCount = locateParticles(pc, ranks, invalid);
1346
1347 IpplTimings::stopTimer(locateTimer);
1348
1349 // 2nd step
1350
1351 // figure out how many receives
1352 static IpplTimings::TimerRef preprocTimer = IpplTimings::getTimer("sendPreprocess");
1353 IpplTimings::startTimer(preprocTimer);
1354
1355 std::vector<size_type> nSends(nRanks, 0);
1356
1357 std::fill(this->nRecvs_m.begin(), this->nRecvs_m.end(), 0);
1358 this->window_m.fence(0);

1360 for (int rank = 0; rank < nRanks; ++rank) {
1361 if (rank == Comm->rank()) {
1362 // we do not need to send to ourselves
1363 continue;
1364 }
1365 nSends[rank] = numberOfSends(rank, ranks);
1366 this->window_m.template put<size_type>(nSends.data() + rank, rank, Comm->rank());
1367 }

1378 for (int rank = 0; rank < nRanks; ++rank) {

1380 hash_type hash("hash", nSends[rank]);
1381 fillHash(rank, ranks, hash);
1382
1383 pc.sendToRank(rank, tag, requests, hash);
1384 }
1385 }
1386 IpplTimings::stopTimer(sendTimer);
1387
1388 // 3rd step
1389 static IpplTimings::TimerRef destroyTimer = IpplTimings::getTimer("particleDestroy");
1390 IpplTimings::startTimer(destroyTimer);
1391
1392 pc.internalDestroy(invalid, invalidCount);
1393 Kokkos::fence();
1394
1395 IpplTimings::stopTimer(destroyTimer);
1396 static IpplTimings::TimerRef recvTimer = IpplTimings::getTimer("particleRecv");
1397 IpplTimings::startTimer(recvTimer);
1398 // 4th step
1399 for (int rank = 0; rank < nRanks; ++rank) {
1400 if (this->nRecvs_m[rank] > 0) {
1401 pc.recvFromRank(rank, tag, this->nRecvs_m[rank]);
1402 }
1403 }
1404 IpplTimings::stopTimer(recvTimer);
1405
1406 IpplTimings::startTimer(sendTimer);
1407
1408 if (requests.size() > 0) {
1409 MPI_Waitall(requests.size(), requests.data(), MPI_STATUSES_IGNORE);
1410 }
1411 IpplTimings::stopTimer(sendTimer);
1412
1413 IpplTimings::stopTimer(ParticleUpdateTimer);
1414 }
