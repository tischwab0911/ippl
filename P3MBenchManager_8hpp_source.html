456 void initializeParticles() {
457 Inform m("Initialize Particles");
458
459 int commSize = ippl::Comm->size();
460 int rank = ippl::Comm->rank();
461
462 static IpplTimings::TimerRef ITimer = IpplTimings::getTimer("initializeParticles");
463 static IpplTimings::TimerRef CTimer = IpplTimings::getTimer("particleCreation");
464 static IpplTimings::TimerRef GTimer = IpplTimings::getTimer("generateCoordinates");
465 static IpplTimings::TimerRef UTimer = IpplTimings::getTimer("updateToRank");
466
467 IpplTimings::startTimer(ITimer);
468
469 unsigned np = this->totalP_m;
470 unsigned nloc = np / commSize;
471
472 this->Q_m = np;
473
474 // make sure all particles are accounted for
475 if(rank == commSize-1){
476 nloc = np - (commSize-1)*nloc;
477 }
478
479 IpplTimings::startTimer(CTimer);
480 this->pcontainer_m->create(nloc);
481 IpplTimings::stopTimer(CTimer);
482
483
484 auto P = this->pcontainer_m->P.getView();
485 auto R = this->pcontainer_m->R.getView();
486 auto Q = this->pcontainer_m->Q.getView();
487
488 auto hLocalRegions = this->pcontainer_m->getLayout().getRegionLayout().gethLocalRegions();
489 Vector_t<T, Dim> domainCentre, domainRadius;
490
491 for(int d = 0; d < Dim; ++d){
492 domainCentre[d] = (hLocalRegions(rank)[d].max() + hLocalRegions(rank)[d].min()) / 2;
493 domainRadius[d] = (hLocalRegions(rank)[d].max() - hLocalRegions(rank)[d].min()) / 2;
494 }
495
496 Kokkos::fence();
497
498 // make sure this runs on the host, device does not work yet
499 Kokkos::Random_XorShift64_Pool<Device> rand_pool((size_type)(42 + 24 * rank));
500
501 IpplTimings::startTimer(GTimer);
502 Kokkos::parallel_for("initialize particles", Kokkos::RangePolicy<Device>(0, nloc),
503 KOKKOS_LAMBDA(const size_t index) {
504 Vector_t<T, Dim> x(0.0);
505
506 auto generator = rand_pool.get_state();
507
508 // obtain random numbers
509 double u = generator.drand();
510 for(int i = 0; i < Dim; ++i){
511 x[i] = generator.normal(0.0, 1.0);
512 }
513
514 rand_pool.free_state(generator);
515
516 // calculate position
517 T normsq = x[0] * x[0] + x[1] * x[1] + x[2] * x[2];
518 Vector_t<T, Dim> pos = domainCentre + domainRadius * (Kokkos::pow(u, 1./3.) / Kokkos::sqrt(normsq)) * x;
519
520 for(int d = 0; d < Dim; ++d){
521 P(index)[d] = 0; // initialize with zero momentum
522 R(index)[d] = pos[d];
523 }
524 Q(index) = 1.0;
525 }
526 );
527
528 // we need to wait for all other ranks to have finished the particle initialization
529 // before we can update them to their corresponding rank
530 Kokkos::fence();
531 ippl::Comm->barrier();
532
533 IpplTimings::stopTimer(GTimer);
534
535 IpplTimings::startTimer(UTimer);
536 this->pcontainer_m->update();
537 IpplTimings::stopTimer(UTimer);
538
539 ippl::Comm->barrier();
540 IpplTimings::stopTimer(ITimer);
541
542 // debug output, can be ignored
543 std::cerr << this->pcontainer_m->getLocalNum() << std::endl;

673 void par2par() override {
674
675 // get particle data
676 auto R = this->pcontainer_m->R.getView();
677 auto E = this->pcontainer_m->E.getView();
678 auto P = this->pcontainer_m->P.getView();
679 auto offset = this->pcontainer_m->getOffset();
680 auto Q = this->pcontainer_m->Q.getView();
681
682 // get simulation specific data
683 auto rcut = this->rcut_m;
684 auto alpha = this->alpha_m;
685 auto epsilon = this->epsilon_m;
686
687 // get neighbor mesh data
688 auto cellStartingIdx = this->pcontainer_m->getNL();
689 size_type totalCells = cellStartingIdx.size() - 1;
690 auto nCells = this->nCells_m;
691 int xCells = nCells[0];
692 int yCells = nCells[1];
693 int zCells = nCells[2];
694
695 assert(totalCells == xCells * yCells * zCells && "Invalid number of cells");
696
697 Kokkos::View<unsigned[1], Device> counter("counter");
698 using team_t = typename Kokkos::TeamPolicy<Device>::member_type;
699
700 // calculate interaction force
701 Kokkos::parallel_for("Particle-Particle", Kokkos::TeamPolicy<Device>(totalCells, Kokkos::AUTO()),
702 KOKKOS_LAMBDA(const team_t& team){
703 const size_type cellIdx = team.league_rank();
704
705 // calculate cellIdx in each dimension
706 int xIdx = cellIdx / (yCells * zCells);
707 int yIdx = (cellIdx % (yCells * zCells)) / zCells;
708 int zIdx = cellIdx % zCells;
709
710 // get number of particles in current cell
711 const size_type start = cellStartingIdx(cellIdx);
712 const size_type end = cellStartingIdx(cellIdx+1);
713 const size_type nParticles = end - start;
714
715 Kokkos::parallel_for(Kokkos::TeamThreadRange(team, 14),
716 [=](const int& neighborIdx){
717
718 // get offset for neighbor cell
719 const int offsetX = offset(neighborIdx * 3 + 0);
720 const int offsetY = offset(neighborIdx * 3 + 1);
721 const int offsetZ = offset(neighborIdx * 3 + 2);
722
723 // check if neighbor is within domain
724 if ((xIdx + offsetX < 0) || (xIdx + offsetX >= xCells) ||
725 (yIdx + offsetY < 0) || (yIdx + offsetY >= yCells) ||
726 (zIdx + offsetZ < 0) || (zIdx + offsetZ >= zCells)) {
727 return;
728 }
729
730 // get number of particles in neighbor cell
731 const size_type neighborCellIdx = (xIdx + offsetX) * yCells * zCells + (yIdx + offsetY) * zCells + (zIdx + offsetZ);
732 const size_type neighborStart = cellStartingIdx(neighborCellIdx);
733 const size_type neighborEnd = cellStartingIdx(neighborCellIdx+1);
734 const size_type nNeighborParticles = neighborEnd - neighborStart;
735
736 auto threadVectorMDRange =
737 Kokkos::ThreadVectorMDRange<Kokkos::Rank<2>, team_t>(team, nParticles, nNeighborParticles);
738
739 Kokkos::parallel_for(threadVectorMDRange,
740 [=](const int& i, const int& j){
741 const size_type ii = start + i;
742 const size_type jj = neighborStart + j;
743 if (((cellIdx == neighborCellIdx) && ii >= jj)) return;
744
745 double rsq_ij = 0.0;
746 Vector_t<T, Dim> dist_ij = R(ii) - R(jj);
747 for (int d = 0; d < Dim; ++d) {
748 rsq_ij += dist_ij[d] * dist_ij[d];
749 }
750
751 double r_ij = Kokkos::sqrt(rsq_ij);
752 if (r_ij >= rcut) return;
753
754 // calculate and apply force
755 Vector_t<T, Dim> F_ij = ke * (dist_ij/r_ij) * ((2.0 * alpha * Kokkos::exp(-alpha * alpha * rsq_ij))/ (Kokkos::sqrt(Kokkos::numbers::pi) * r_ij) + (1.0 - Kokkos::erf(alpha * r_ij)) / rsq_ij);
756 Kokkos::atomic_sub(&E(ii), F_ij * Q(jj));
757 Kokkos::atomic_add(&E(jj), F_ij * Q(ii));
758 }
759 );
760 }
761 );
762 }
763 );
764
765 // Kokkos::fence();
766 // ippl::Comm->barrier();
767
768 std::cerr << "Particle-Particle Interaction finished" << std::endl;
769 }
