216 Inform m("Initialize Particles");
217
218 int commSize = ippl::Comm->size();
219 int rank = ippl::Comm->rank();
220
221 static IpplTimings::TimerRef ITimer = IpplTimings::getTimer("initializeParticles");
222 static IpplTimings::TimerRef CTimer = IpplTimings::getTimer("particleCreation");
223 static IpplTimings::TimerRef GTimer = IpplTimings::getTimer("generateCoordinates");
224 static IpplTimings::TimerRef UTimer = IpplTimings::getTimer("updateToRank");

226 IpplTimings::startTimer(ITimer);
227
228 unsigned np = this->totalP_m;
229 unsigned nloc = np / commSize;
230
231 this->Q_m = np;
232
233 // make sure all particles are accounted for
234 if(rank == commSize-1){
235 nloc = np - (commSize-1)*nloc;
236 }
237
238 IpplTimings::startTimer(CTimer);
239 this->pcontainer_m->create(nloc);
240 IpplTimings::stopTimer(CTimer);
241
242
243 auto P = this->pcontainer_m->P.getView();
244 auto R = this->pcontainer_m->R.getView();
245 auto Q = this->pcontainer_m->Q.getView();
246
247 double beamRad = this->beamRad_m;
248
249 Kokkos::fence();
250
251 // make sure this runs on the host, device does not work yet
252 Kokkos::Random_XorShift64_Pool<Device> rand_pool((size_type)(42 + 24 * rank));
253
254 IpplTimings::startTimer(GTimer);
255 Kokkos::parallel_for("initialize particles", Kokkos::RangePolicy<Device>(0, nloc),
256 KOKKOS_LAMBDA(const size_t index) {
257 Vector_t<T, Dim> x(0.0);
258
259 auto generator = rand_pool.get_state();
260
261 // obtain random numbers
262 double u = generator.drand();
263 for(int i = 0; i < Dim; ++i){
264 x[i] = generator.normal(0.0, 1.0);
265 }
266
267 rand_pool.free_state(generator);
268
269 // calculate position
270 T normsq = x[0] * x[0] + x[1] * x[1] + x[2] * x[2];
271 Vector_t<T, Dim> pos = beamRad * (Kokkos::pow(u, 1./3.) / Kokkos::sqrt(normsq)) * x;
272
273 for(int d = 0; d < Dim; ++d){
274 P(index)[d] = 0; // initialize with zero momentum
275 R(index)[d] = pos[d];
276 }
277 Q(index) = 1.0;
278 }
279 );
280
281 // we need to wait for all other ranks to have finished the particle initialization
282 // before we can update them to their corresponding rank
283 Kokkos::fence();
284 ippl::Comm->barrier();
285
286 IpplTimings::stopTimer(GTimer);
287
288 IpplTimings::startTimer(UTimer);
289 this->pcontainer_m->update();
290 IpplTimings::stopTimer(UTimer);
291
292 ippl::Comm->barrier();
293 IpplTimings::stopTimer(ITimer);
294
295 // debug output, can be ignored
296 std::cerr << this->pcontainer_m->getLocalNum() << std::endl;

592 auto R = this->pcontainer_m->R.getView();

600 auto alpha = this->alpha_m;
601 auto epsilon = this->epsilon_m;

603 // get neighbor mesh data
604 auto cellStartingIdx = this->pcontainer_m->getNL();
605 size_type totalCells = cellStartingIdx.size() - 1;
606 auto nCells = this->nCells_m;
607 int xCells = nCells[0];

609 int zCells = nCells[2];

611 assert(totalCells == xCells * yCells * zCells && "Invalid number of cells");
612
613 Kokkos::View<unsigned[1], Device> counter("counter");
614 using team_t = typename Kokkos::TeamPolicy<Device>::member_type;

616 // calculate interaction force
617 Kokkos::parallel_for("Particle-Particle", Kokkos::TeamPolicy<Device>(totalCells, Kokkos::AUTO()),
618 KOKKOS_LAMBDA(const team_t& team){
619 const size_type cellIdx = team.league_rank();
620
621 // calculate cellIdx in each dimension
622 int xIdx = cellIdx / (yCells * zCells);
623 int yIdx = (cellIdx % (yCells * zCells)) / zCells;
624 int zIdx = cellIdx % zCells;
625
626 // get number of particles in current cell
627 const size_type start = cellStartingIdx(cellIdx);
628 const size_type end = cellStartingIdx(cellIdx+1);
629 const size_type nParticles = end - start;

632 [=](const int& neighborIdx){
633
634 // get offset for neighbor cell
635 const int offsetX = offset(neighborIdx * 3 + 0);
636 const int offsetY = offset(neighborIdx * 3 + 1);
637 const int offsetZ = offset(neighborIdx * 3 + 2);
638
639 // check if neighbor is within domain
640 if ((xIdx + offsetX < 0) || (xIdx + offsetX >= xCells) ||
641 (yIdx + offsetY < 0) || (yIdx + offsetY >= yCells) ||
642 (zIdx + offsetZ < 0) || (zIdx + offsetZ >= zCells)) {
643 return;
644 }
645
646 // get number of particles in neighbor cell
647 const size_type neighborCellIdx = (xIdx + offsetX) * yCells * zCells + (yIdx + offsetY) * zCells + (zIdx + offsetZ);
648 const size_type neighborStart = cellStartingIdx(neighborCellIdx);
649 const size_type neighborEnd = cellStartingIdx(neighborCellIdx+1);
650 const size_type nNeighborParticles = neighborEnd - neighborStart;
651
652 auto threadVectorMDRange =
653 Kokkos::ThreadVectorMDRange<Kokkos::Rank<2>, team_t>(team, nParticles, nNeighborParticles);
654
655 Kokkos::parallel_for(threadVectorMDRange,
656 [=](const int& i, const int& j){
657 const size_type ii = start + i;
658 const size_type jj = neighborStart + j;
659 if (((cellIdx == neighborCellIdx) && ii >= jj)) return;
660
661 double rsq_ij = 0.0;
662 Vector_t<T, Dim> dist_ij = R(ii) - R(jj);
663 for (int d = 0; d < Dim; ++d) {
664 rsq_ij += dist_ij[d] * dist_ij[d];
665 }
666
667 double r_ij = Kokkos::sqrt(rsq_ij);
668 if (r_ij >= rcut) return;
669
670 // calculate and apply force
671 Vector_t<T, Dim> F_ij = ke * (dist_ij/r_ij) * ((2.0 * alpha * Kokkos::exp(-alpha * alpha * rsq_ij))/ (Kokkos::sqrt(Kokkos::numbers::pi) * r_ij) + (1.0 - Kokkos::erf(alpha * r_ij)) / rsq_ij);
672 Kokkos::atomic_sub(&E(ii), F_ij * Q(jj));
673 Kokkos::atomic_add(&E(jj), F_ij * Q(ii));
674 }
675 );
676 }
677 );
678 }
679 );
680
681 // Kokkos::fence();
682 // ippl::Comm->barrier();
683
684 std::cerr << "Particle-Particle Interaction finished" << std::endl;
