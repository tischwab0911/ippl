8 TruncatedGreenParticleInteraction<ParticleContainer, ScalarAttribute,

10 Scalar_t r2, Scalar_t alpha,
11 Scalar_t forceConstant,
12 Scalar_t qm2) {
13 const Scalar_t r = Kokkos::sqrt(r2);
14
15 // F = - q * forceConstant grad [(1 - erf(alpha * r)) / r].

18 / (Kokkos::sqrt(Kokkos::numbers::pi) * r)
19 + (1.0 - Kokkos::erf(alpha * r)) / r2);
20 }
